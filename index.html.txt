    <script>
        // تم تحديث رقم الواتساب الخاص بك هنا بنجاح
        const WHATSAPP_NUMBER = "963982691478"; 

        // بيانات المنيو
        const menuProducts = [
            { id: 1, name: "مشاوي لحم مشكلة (كباب وشقف)", price: 180000, unit: "كيلو" },
            { id: 2, name: "مشاوي دجاج مشكلة (كباب وشقف)", price: 100000, unit: "كيلو" },
            { id: 3, name: "سودة دجاج مشوية", price: 90000, unit: "كيلو" },
            { id: 4, name: "سودة غنم مشوية", price: 160000, unit: "كيلو" },
            { id: 5, name: "لبن عيران", price: 5000, unit: "عدد" },
            { id: 6, name: "بيبسي", price: 7000, unit: "عدد" }
        ];

        let cart = [];

        // عرض المنيو في الصفحة
        function displayMenu() {
            const menuContainer = document.getElementById('menu-container');
            menuContainer.innerHTML = menuProducts.map(product => `
                <div class="menu-item">
                    <div>
                        <div class="item-name">${product.name}</div>
                        <div class="item-price">${product.price.toLocaleString()} ل.س / ${product.unit}</div>
                    </div>
                    <button class="add-btn" onclick="addToCart(${product.id})">إضافة للسلة</button>
                </div>
            `).join('');
        }

        // إضافة عنصر للسلة
        function addToCart(id) {
            const product = menuProducts.find(p => p.id === id);
            const cartItem = cart.find(item => item.id === id);

            if (cartItem) {
                cartItem.quantity++;
            } else {
                cart.push({ ...product, quantity: 1 });
            }
            updateCart();
        }

        // تغيير الكمية في السلة
        function changeQuantity(id, delta) {
            const cartItem = cart.find(item => item.id === id);
            if (cartItem) {
                cartItem.quantity += delta;
                if (cartItem.quantity <= 0) {
                    cart = cart.filter(item => item.id !== id);
                }
            }
            updateCart();
        }

        // تحديث واجهة السلة والمجموع
        function updateCart() {
            const cartContainer = document.getElementById('cart-container');
            const totalPriceEl = document.getElementById('total-price');
            
            if (cart.length === 0) {
                cartContainer.innerHTML = `<p style="color: var(--gray); text-align: center;">السلة فارغة حالياً</p>`;
                totalPriceEl.innerText = "0 ل.س";
                return;
            }

            let total = 0;
            cartContainer.innerHTML = cart.map(item => {
                total += item.price * item.quantity;
                return `
                    <div class="cart-item">
                        <div class="cart-item-info">
                            <div><strong>${item.name}</strong></div>
                            <div style="color: var(--gold); font-size: 14px;">${(item.price * item.quantity).toLocaleString()} ل.س</div>
                        </div>
                        <div class="quantity-controls">
                            <button class="qty-btn" onclick="changeQuantity(${item.id}, -1)">-</button>
                            <span>${item.quantity}</span>
                            <button class="qty-btn" onclick="changeQuantity(${item.id}, 1)">+</button>
                        </div>
                    </div>
                `;
            }).join('');

            totalPriceEl.innerText = total.toLocaleString() + " ل.س";
        }

        // إرسال الطلب عبر الواتساب
        function sendToWhatsApp() {
            const name = document.getElementById('customer-name').value.trim();
            const address = document.getElementById('customer-address').value.trim();

            if (cart.length === 0) {
                alert("يرجى إضافة بعض الأصناف إلى السلة أولاً!");
                return;
            }
            if (!name || !address) {
                alert("يرجى ملء الاسم والعنوان لإتمام الطلب.");
                return;
            }

            // تجهيز نص الرسالة للواتساب
            let message = `*طلب جديد من موقع مشويات أبو الخير* 🔥\n\n`;
            message += `👤 *الاسم:* ${name}\n`;
            message += `📍 *العنوان:* ${address}\n\n`;
            message += `🛒 *الطلبات:*\n`;

            let total = 0;
            cart.forEach(item => {
                const itemTotal = item.price * item.quantity;
                total += itemTotal;
                message += `- ${item.name} (الكمية: ${item.quantity}) -> ${itemTotal.toLocaleString()} ل.س\n`;
            });

            message += `\n💰 *المجموع الإجمالي:* ${total.toLocaleString()} ل.س`;

            // ترميز النص ليتوافق مع الرابط
            const encodedMessage = encodeURIComponent(message);
            const whatsappUrl = `https://wa.me/${WHATSAPP_NUMBER}?text=${encodedMessage}`;

            // فتح الواتساب في نافذة جديدة
            window.open(whatsappUrl, '_blank');
        }

        // تشغيل عرض المنيو عند فتح الصفحة
        displayMenu();
    </script>
